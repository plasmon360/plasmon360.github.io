def expt_run():
    num_of_tosses_in_each_trial = [500, 1000, 2000, 5000, 7500, 10000, 12500, 15000, 17500, 20000]
    fraction_of_heads_in_each_trial = []
    for trial in num_of_tosses_in_each_trial:
        tosses = [mycoin.toss for toss in range(trial)]
        fraction_of_heads_in_each_trial.append(sum(tosses)/trial)
    return num_of_tosses_in_each_trial, fraction_of_heads_in_each_trial

import matplotlib.pyplot as plt 
import matplotlib
import numpy as np 

font = {'family' : 'Tahoma',
        'weight' : 'bold',
        'size'   : 16}

matplotlib.rc('font', **font)


class Coin():
    
    '''
    Coin that can be 
    
    1) Tossed once with a certain probability of landing a heads
    2) Generate a sample (results of Heads/Tails in a series of tossing).
    3) For generated sample, calculate the stats about it
    
    '''
    def __init__(self, heads_probability = 1/2):
        
        # heads_probabilty defines the probability that it will land heads
        # 1-heads_probability defines the probability that it lands tails
        
        self.heads_prob = heads_probability
    
    def toss(self):
        # Return a random varible that could be Heads (1) or Tails (0)
        
        return np.random.randint(0,2)
    
    def generate_sample_of_tosses_and_its_stats(self, num_tosses_in_sample = 500):
        # This generates a sample. Each sample consists of 'num_tosses_in_sample' tosses
        # of our coin.
        
        # Lets store everything related to one sample in this dict
        tosses = {}
        
        # Store num of tosses in this sample
        tosses['Num of tosses'] = num_tosses_in_sample
        
        # These are the all the tosses (Heads/1 and Tails/0) in this sample
        tosses['Sample'] = [self.toss() for toss in range(num_tosses_in_sample)]
        
        # Sample sum here indicates the total number of heads in all over tosses in this sample, because 
        # Tails are represented by 1 and Heads are represented by 0
        
        tosses['Sample_Sum/Num_Heads']= np.sum(tosses['Sample'])
        
        # Sample Standard deviation of sample
        tosses['Sample_Standard_Deviation'] = np.std(tosses['Sample'])*(tosses['Num of tosses']/(tosses['Num of tosses']-1))
        
        # Standard error in the number of heads = sqrt(num of tosses) * standard device
        tosses['Sample_Standard_Error_Sum'] = np.sqrt(tosses['Num of tosses']) * tosses['Sample_Standard_Deviation']
        
        # 95% confidence intervals for the sample
        tosses['Sample_Confidence_Interval_width'] = 2*tosses['Sample_Standard_Error_Sum']
        
        # Expected number of heads or Expected sum of random variable
        tosses['Expected_Sum']=  tosses['Num of tosses'] * self.heads_prob
        
        # Sample Z-score = (Observed Heads - Expected Heads)/(Standard Error of num of Heads)
        tosses['Sample_Z-score'] = (tosses['Sample_Sum/Num_Heads'] - tosses['Expected_Sum'])/tosses['Sample_Standard_Error_Sum']
        
        return tosses

# Lets do an experiment.

# Lets take a fair coin
fair_coin= Coin(heads_probability=0.5)

# lets take many samples. Remember, a sample consists of series of tosses and generating sample stats
num_samples = 100

# lets pick number of tosses in each sample
tosses_in_each_sample = 500

# Set up over plot figure
fig, ax = plt.subplots(nrows= 2, figsize=(8, 8*1.616 ))

# lets start generating samples
for index, sample in enumerate(np.arange(num_samples)):
    
    # generate a series of tosses and calculates following stats:
    
    # Sample_Sum: Indicates how many heads will be in that sample of tosses
    
    # Expected sum : how many heads will be there on average without randomness
    
    # Standard deviation of our sample. Note that it also turns out to be the standard deviation of our random variable (0 or 1)
    # which is sqrt(heads_prob * (1-heads_prob))
    
    # Standard Error of Sum: VERY IMPORTANT: This is the standard deviation of sum of random variable and 
    ## and not the random variable.
    
    # Confidence Interval width: We will get to this at the end
    
    
    sample = fair_coin.generate_sample_of_tosses_and_its_stats(tosses_in_each_sample)
        
    # If the Expected number of Heads is the within the 95% confidence interval
    
    if (sample['Sample_Sum/Num_Heads'] - sample['Sample_Confidence_Interval_width'] 
        < sample['Expected_Sum'] <
        sample['Sample_Sum/Num_Heads'] + sample['Sample_Confidence_Interval_width'] ):        
        
        color = 'g'
    else:
        color = 'r'
    
    # 
    ax[0].errorbar(x=index, y=sample['Sample_Sum/Num_Heads'],
                   yerr = sample['Sample_Confidence_Interval_width'],capsize=5,
                   fmt='o', color = color)    
    
    ax[1].plot(index, sample['Sample_Z-score'],'o',color = color)
    
ax[1].set_xlabel('Sample number')
ax[0].set_xlabel('Sample number')
ax[1].set_ylabel('Z Score')
ax[0].set_ylabel('Observed Heads')
ax[0].set_title('Observed Heads with 95% confidence intervals')
ax[1].hlines(y=2, xmin=0, xmax=num_samples,color='r')
ax[1].hlines(y=-2, xmin=0, xmax=num_samples, color = 'r')
ax[1].hlines(y=0, xmin=0, xmax=num_samples, color = 'g')
ax[0].hlines(y= fair_coin.heads_prob * tosses_in_each_sample, xmin=0, xmax=num_samples, color = 'b')
